# DFS code Simple 


# Depth First Search (DFS) Algorithm using Recursion
# For an Undirected Tree

# Recursive DFS function
def dfs(tree, node, visited):
    print(node, end=" ")
    visited[node] = True

    # Traverse all connected (child) nodes
    for child in tree[node]:
        if not visited[child]:
            dfs(tree, child, visited)


# --- MAIN PROGRAM ---
print("Depth First Search (DFS) for a Tree Data Structure\n")

# Representing the tree as an adjacency list
# Example Tree:
#        0
#       / \
#      1   2
#     / \
#    3   4

tree = {
    0: [1, 2],
    1: [0, 3, 4],
    2: [0],
    3: [1],
    4: [1]
}

# Number of nodes
n = len(tree)

# Keep track of visited nodes
visited = [False] * n

# Start DFS from the root node (0)
print("DFS Traversal starting from root node 0:")
dfs(tree, 0, visited)
print()





# DFS Code With Input  

# Depth First Search using Recursion in an Undirected Graph
# Depth First Search (DFS) for a Tree Data Structure
# Accepting Tree as Input (Undirected Tree)

# Recursive DFS function
def dfs(tree, node, visited):
    print(node, end=" ")
    visited[node] = True

    # Visit all adjacent (child) nodes
    for child in tree[node]:
        if not visited[child]:
            dfs(tree, child, visited)


# --- MAIN PROGRAM ---
print("Depth First Search (DFS) for a Tree Data Structure\n")

# Input number of nodes
n = int(input("Enter the number of nodes in the tree: "))

# Initialize adjacency list for the tree
tree = {i: [] for i in range(n)}

# Since a tree has (n - 1) edges
print("\nEnter the edges of the tree (u v):")
for i in range(n - 1):
    u, v = map(int, input(f"Edge {i+1}: ").split())
    tree[u].append(v)
    tree[v].append(u)

# Input root node
root = int(input("\nEnter the root node to start DFS: "))

# Keep track of visited nodes
visited = [False] * n

# Perform DFS traversal
print("\nDFS Traversal of the tree starting from root node", root, ":")
dfs(tree, root, visited)
print()
